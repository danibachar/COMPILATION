int IsPrime(int p)
{
    int i := 2;
    int j := 2;

    while (i<p)
    {
        while (j<p)
        {
            if (i*j = p)
            {
                 return 1;
            }
            j := j+1;
        }
        i := i+1;
    }
    return 0;
}

{

    while (p < end+1)
    {

}
